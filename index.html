<!-- HTML -->
<form id="contactForm">
  <dfr-text-question 
    id="fullName" 
    label="Full Name" 
    required>
  </dfr-text-question>

  <dfr-select-question 
    id="companySize" 
    label="Company Size">
  </dfr-select-question>
</form>

<!-- JavaScript -->
<script>
  const sizeSelect = document.getElementById('companySize');
  
  // Set options dynamically
  sizeSelect.question = {
    options: [
      { label: '1-10', value: 'small' },
      { label: '11-50', value: 'medium' },
      { label: '50+', value: 'large' }
    ]
  };

  // Listen for changes
  document.addEventListener('value-changed', (e) => {
    console.log(e.detail.value);
  });
</script>
